function fizzBuzz(n: number): string[] {
    const h: string[] = [];
    for (let i = 1; i <= n; i++) {
        if (i % 15 == 0)h.push("FizzBuzz" );
        else if  (i % 3 ==0) h.push( "Fizz");
        
        else if  (i % 5== 0) h.push("Buzz" );
        else h.push(String(i));
    }
    return h;
}
